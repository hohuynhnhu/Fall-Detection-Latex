\section{Cài đặt hệ thống}

\subsection{Cài đặt Desktop Application}

\begin{enumerate}
    \item Clone repository và tạo môi trường ảo Python
    \item Cài đặt dependencies: pip install -r requirements.txt
    \item Download pretrained models (YOLOv8, CNN, Siamese, DeepFace)
    \item Cấu hình file config.yaml với thông tin camera và API
    \item Chạy: python run.py
\end{enumerate}

\subsection{Cài đặt Backend Server}

\begin{enumerate}
    \item Clone repository backend
    \item Cấu hình file .env với MongoDB, Firebase, Cloudinary
    \item Cài đặt: pip install -r requirements.txt
   
\end{enumerate}

\subsection{Cài đặt Mobile Application}

\begin{enumerate}
    \item Clone repository và mở trong Android Studio
    \item Đồng bộ Gradle dependencies
    \item Build: ./gradlew installDebug hoặc ./gradlew assembleRelease
\end{enumerate}

\textbf{Phân tích kết quả}

Thống kê tổng hợp:
\begin{itemize}
    \item Tổng số lần test: 100 (10 cặp ảnh $\times$ 10 lần)
    \item Tổng số nhận dạng đúng: 98/100
    \item Độ chính xác trung bình: 98.00\%
    \item Tỷ lệ lỗi: 2.00\%
    \item Số lần test đạt 100\%: 7/10 lần (70\%)
    \item Số lần test có lỗi: 3/10 lần (30\%)
\end{itemize}

Độ ổn định của mô hình:
\begin{itemize}
    \item Mô hình cho kết quả ổn định qua 10 lần kiểm thử
    \item Accuracy dao động trong khoảng từ 90\% đến 100\%
    \item Các lỗi xuất hiện không ngẫu nhiên mà tập trung ở một số cặp ảnh nhất định
\end{itemize}

Phân tích lỗi:
\begin{itemize}
    \item Số cặp ảnh gây lỗi: 1 cặp trong 10 cặp ảnh
    \item Nguyên nhân chủ yếu:
    \begin{itemize}
        \item Chất lượng ảnh thấp (mờ, thiếu sáng)
        \item Góc chụp khác biệt 
    \end{itemize}
    \item Distance score của các trường hợp lỗi nằm gần ngưỡng quyết định (threshold)
\end{itemize}

Đánh giá:
\begin{itemize}
    \item \textbf{Ưu điểm:}
    \begin{itemize}
        \item Độ chính xác cao (98.00\%), đáp ứng yêu cầu hệ thống (≥98\%)
        \item Hoạt động ổn định, kết quả lặp lại tốt qua nhiều lần test
        \item Thời gian xử lý nhanh, trung bình khoảng 0.5–1 giây cho mỗi cặp ảnh
    \end{itemize}
\end{itemize}

Mô hình DeepFace sử dụng Facenet512 đạt độ chính xác 98.00\% trên 100 lượt kiểm thử với 10 cặp ảnh, đáp ứng yêu cầu đề ra của hệ thống. Kết quả cho thấy mô hình hoạt động ổn định và đủ tin cậy để triển khai trong môi trường thực tế. Tuy nhiên, cần mở rộng thêm tập dữ liệu kiểm thử để đánh giá toàn diện hơn trong các điều kiện đa dạng.
